Bài này khá giống bài khiêu vũ

```cpp
#include<bits/stdc++.h>

using namespace std;
typedef long long ll;
int main(){
    int n, m,k; cin >> n >> m >> k;
    int a[n];
    for(int &x : a){
        cin >> x;
    }
    int b[m];
    for(int &x : b){
        cin >> x;
    }
    sort(a, a + n);sort(b, b + m);
    int i = 0,j = 0, ans = 0;
    while(i < n && j < m){
        if(abs(a[i] - b[j]) <= k){
            ++ans;++i;++j;
        }else if(a[i] - b[j] < k){
            ++j;
        }else{
            ++i;
        }
    }
    cout << ans << endl;
}
```
