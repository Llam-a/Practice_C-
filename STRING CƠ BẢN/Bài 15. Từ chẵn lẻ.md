#include<bits/stdc++.h>

using namespace std;

int main(){
    string s;
    getline(cin, s);
    stringstream ss(s);
    string token;
    int dem = 0;
    while(ss >> token) {
        ++dem;
        if(dem % 2 == 0){
            reverse(token.begin(), token.end());
        }
        cout << token << " ";
    }
}
